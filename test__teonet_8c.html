int main() {
KSN_SET_TEST_MODE(1);
// Initialize the CUnit test registry
if (CUE_SUCCESS != CU_initialize_registry())
return CU_get_error();
// Add a suite to the registry
pSuite = CU_add_suite("Teonet library Crypt module", init_suite, clean_suite);
if (NULL == pSuite) {
CU_cleanup_registry();
return CU_get_error();
}
add_suite_1_tests();
